}

                           passwd=db_config['passwd'], charset=db_config['charset'])
　
　　# サンプルデ`タの追加

　　# プレ`スホルダの使用例
　　# 1つの龊悉摔献钺幛 , がないとエラ`。('JOHNSON') ではなく ('JOHNSON',)
　　cursor.execute("INSERT INTO students VALUES (2, %s)", ('JOHNSON',))
　　cursor.execute("INSERT INTO students VALUES (%s, %s)", (3, 'WILLIAMS'))
　　# 名前付きプレ`スホルダ
　　cursor.execute("INSERT INTO students VALUES (%(id)s, %(name)s)", {'id': 4, 'name': 'JONES'})
 
　　# }数レコ`ドを一度に啡 executemany メソッドを使用
　　persons = [
    　(5, 'BROWN'),
    　(6, 'DAVIS'),
　　]
　　
　　cursor.executemany("INSERT INTO students VALUES (%s, %s)", persons)
 
　　# コミット（デ`タ保存をg行）
　　# commit() を忘れてると保存されないことに注意
　　conn.commit()
except MySQLdb.Error as ex:
print('MySQL Error: ', ex)

import MySQLdb
 
def db_sample():
    """ 接Aサンプル """
 
    # 接Aする 
    con = MySQLdb.connect(
            user='db user',
            passwd='db password',
            host='localhost',
            db='sample')
 
    # カ`ソルを取得する
    cur= con.cursor()
     
    # クエリをg行する
    sql = "select id, body, post_code, created from posts"
    cur.execute(sql)
 
    # g行Y果をすべて取得する
    rows = cur.fetchall()
     
    # 一行ずつ表示する
    for row in rows:
        print(row)
 
    cur.close
    con.close
 
if __name__ == "__main__":
    db_sample()
